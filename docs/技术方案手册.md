#### Excel 解析实现

```typescript
import * as XLSX from 'xlsx'

interface Ticket {
  issueType: string
  key: string
  domain: string
  summary: string
  assignee: string
  status: string
  createDate: Date
  projectName: string
  priority: string
  customerIssueType: string
  dueDate: Date
  duration: number
}

export async function parseExcel(file: File): Promise<Ticket[]> {
  return new Promise((resolve, reject) => {
    const reader = new FileReader()

    reader.onload = (e) => {
      try {
        const data = e.target.result
        const workbook = XLSX.read(data, { type: 'array' })

        // 获取第一个工作表
        const sheetName = workbook.SheetNames[0]
        const worksheet = workbook.Sheets[sheetName]

        // 转换为 JSON
        const jsonData = XLSX.utils.sheet_to_json(worksheet)

        // 数据清洗和转换
        const tickets = jsonData.map(row => transformTicket(row))

        resolve(tickets)
      } catch (error) {
        reject(error)
      }
    }

    reader.onerror = () => reject(new Error('文件读取失败'))
    reader.readAsArrayBuffer(file)
  })
}

function transformTicket(row: any): Ticket {
  return {
    issueType: row['问题类型'] || '',
    key: row['关键字'] || '',
    domain: row['领域模块'] || '',
    summary: row['概要'] || '',
    assignee: row['经办人'] || '',
    status: row['状态'] || '',
    createDate: parseDate(row['创建日期']),
    projectName: row['项目名称'] || '',
    priority: row['紧急程度'] || '',
    customerIssueType: row['客户问题类型'] || '',
    dueDate: parseDate(row['到期日']),
    duration: calculateDuration(row['创建日期'], row['到期日'])
  }
}

function parseDate(dateStr: string): Date {
  if (!dateStr) return null
  // 处理不同的日期格式
  // 格式1: "2026-03-31 17:44"
  // 格式2: "2026/03/31 17:44"
  // 格式3: Excel 序列号

  return new Date(dateStr)
}

function calculateDuration(createDate: string, dueDate: string): number {
  if (!createDate || !dueDate) return 0
  const start = new Date(createDate).getTime()
  const end = new Date(dueDate).getTime()
  return Math.ceil((end - start) / (1000 * 60 * 60 * 24))
}
```

### 3.2 数据分析算法

#### 核心指标计算

```typescript
export interface BasicStats {
  totalCount: number           // 工单总量
  completedCount: number      // 已完成数量
  completionRate: number     // 完成率
  avgDuration: number         // 平均处理时长
  medianDuration: number      // 中位数时长
  activeAssignees: number     // 活跃经办人数
  activeProjects: number      // 活跃项目数
  pendingCount: number        // 待处理数量
}

export function calculateBasicStats(tickets: Ticket[]): BasicStats {
  const totalCount = tickets.length

  const completedCount = tickets.filter(t =>
    t.status === '支持确认完成' || t.status === '研发已完成'
  ).length

  const durations = tickets
    .map(t => t.duration)
    .filter(d => d > 0)
    .sort((a, b) => a - b)

  const avgDuration = durations.reduce((a, b) => a + b, 0) / durations.length
  const medianDuration = durations[Math.floor(durations.length / 2)]

  const uniqueAssignees = new Set(tickets.map(t => t.assignee))
  const uniqueProjects = new Set(tickets.map(t => t.projectName))

  const pendingCount = tickets.filter(t => t.status === '待分析').length

  return {
    totalCount,
    completedCount,
    completionRate: (completedCount / totalCount) * 100,
    avgDuration,
    medianDuration,
    activeAssignees: uniqueAssignees.size,
    activeProjects: uniqueProjects.size,
    pendingCount
  }
}
```

#### 时间趋势分析

```typescript
export interface TrendData {
  dates: string[]      // 日期数组
  created: number[]    // 创建数量
  completed: number[]   // 完成数量
}

export function calculateTrend(
  tickets: Ticket[],
  granularity: 'day' | 'week' | 'month'
): TrendData {
  const groups = new Map<string, { created: number; completed: number }>()

  tickets.forEach(ticket => {
    const key = formatDateByGranularity(ticket.createDate, granularity)

    if (!groups.has(key)) {
      groups.set(key, { created: 0, completed: 0 })
    }

    const group = groups.get(key)
    group.created++

    if (ticket.status === '支持确认完成' || ticket.status === '研发已完成') {
      group.completed++
    }
  })

  // 排序并转换为数组
  const sortedKeys = Array.from(groups.keys()).sort()
  const dates = sortedKeys
  const created = sortedKeys.map(k => groups.get(k).created)
  const completed = sortedKeys.map(k => groups.get(k).completed)

  return { dates, created, completed }
}

function formatDateByGranularity(date: Date, granularity: string): string {
  const d = new Date(date)

  switch (granularity) {
    case 'day':
      return d.toISOString().split('T')[0] // "2026-03-31"
    case 'week':
      const weekStart = getWeekStart(d)
      return weekStart.toISOString().split('T')[0]
    case 'month':
      return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`
    default:
      return d.toISOString().split('T')[0]
  }
}
```

#### 分组统计

```typescript
export interface GroupResult {
  name: string
  value: number
  percentage: number
  details?: any
}

export function groupByDimension(
  tickets: Ticket[],
  dimension: 'assignee' | 'domain' | 'priority' | 'customerIssueType' | 'projectName'
): GroupResult[] {
  const groups = new Map<string, number>()

  tickets.forEach(ticket => {
    const key = ticket[dimension]
    groups.set(key, (groups.get(key) || 0) + 1)
  })

  const total = tickets.length
  const result = Array.from(groups.entries())
    .map(([name, value]) => ({
      name,
      value,
      percentage: (value / total) * 100
    }))
    .sort((a, b) => b.value - a.value)

  return result
}
```

```vue
import { ref, onMounted, onUnmounted, watch } from 'vue'
import * as echarts from 'echarts'
import type { EChartsOption } from 'echarts'

interface Props {
  option: EChartsOption
  width?: string
  height?: string
  theme?: string

const props = withDefaults(defineProps<Props>(), {
  width: '100%',
  height: '400px',
  theme: 'dark'

const chartRef = ref<HTMLElement>()
let chartInstance: echarts.ECharts | null = null

onMounted(() => {
  initChart()
})

onUnmounted(() => {
  disposeChart()
})

watch(() => props.option, (newOption) => {
  if (chartInstance) {
    chartInstance.setOption(newOption, true)
  }
}, { deep: true })

function initChart() {
  if (!chartRef.value) return

  chartInstance = echarts.init(chartRef.value, props.theme)
  chartInstance.setOption(props.option)

  // 响应式
  window.addEventListener('resize', handleResize)
}

function handleResize() {
  chartInstance?.resize()
}

function disposeChart() {
  if (chartInstance) {
    chartInstance.dispose()
    window.removeEventListener('resize', handleResize)
  }
}

defineExpose({
  getInstance: () => chartInstance,
  resize: () => chartInstance?.resize()
})
</script>
```

```typescript
// src/composables/useChartClick.ts

import type { Ticket } from '@/types'
import { useDataStore } from '@/stores/dataStore'
import { format } from 'date-fns'

export interface FilterCondition {
  field: string      // 筛选字段
  operator: string   // eq, in, between, month
  value: any
  description: string
}

export function useChartClick() {
  const dataStore = useDataStore()

  /**
   * 构建筛选条件
   */
  function buildFilterCondition(
    chartType: string,
    clickData: any,
    chartMeta?: any
  ): FilterCondition[] {
    const conditions: FilterCondition[] = []

    switch (chartType) {
      case 'bar':
        // 柱状图：按维度值筛选
        if (clickData.name) {
          const field = chartMeta?.filterField || 'assignee'
          conditions.push({
            field,
            operator: 'eq',
            value: clickData.name,
            description: `${getFieldName(field)} = ${clickData.name}`
          })
        }
        break

      case 'pie':
        // 饼图：按类型筛选
        if (clickData.name) {
          const field = chartMeta?.filterField || 'customerIssueType'
          conditions.push({
            field,
            operator: 'eq',
            value: clickData.name,
            description: `${getFieldName(field)} = ${clickData.name}`
          })
        }
        break

      case 'line':
        // 折线图：按月份筛选
        if (clickData.name) {
          conditions.push({
            field: 'createDate',
            operator: 'month',
            value: clickData.name,
            description: `创建月份 = ${clickData.name}`
          })
        }
        break

      case 'scatter':
        // 散点图：按工单ID筛选
        if (clickData.data && clickData.data.length >= 3) {
          conditions.push({
            field: 'key',
            operator: 'eq',
            value: clickData.data[2],
            description: `工单 = ${clickData.data[2]}`
          })
        }
        break

      case 'heatmap':
        // 热力图：双维度筛选
        const xValue = chartMeta?.xAxisData?.[clickData.data[0]]
        const yValue = chartMeta?.yAxisData?.[clickData.data[1]]

        if (xValue && yValue && chartMeta) {
          conditions.push({
            field: chartMeta.xField,
            operator: 'eq',
            value: xValue,
            description: `${getFieldName(chartMeta.xField)} = ${xValue}`
          })
          conditions.push({
            field: chartMeta.yField,
            operator: 'eq',
            value: yValue,
            description: `${getFieldName(chartMeta.yField)} = ${yValue}`
          })
        }
        break
    }

    return conditions
  }

  /**
   * 过滤工单
   */
  function filterTickets(conditions: FilterCondition[]): Ticket[] {
    return dataStore.filteredData.filter(ticket => {
      return conditions.every(condition => {
        const fieldValue = ticket[condition.field as keyof Ticket]

        switch (condition.operator) {
          case 'eq':
            return fieldValue === condition.value

          case 'month':
            if (fieldValue instanceof Date) {
              return format(fieldValue, 'yyyy-MM') === condition.value
            }
            return false

          default:
            return true
        }
      })
    })
  }

  return {
    buildFilterCondition,
    filterTickets
  }
}

function getFieldName(field: string): string {
  const map: Record<string, string> = {
    assignee: '经办人',
    customerIssueType: '客户问题类型',
    domain: '领域模块',
    status: '状态',
    priority: '紧急程度',
    projectName: '项目名称'
  }
  return map[field] || field
}
```

```typescript
function chunkProcess<T>(
  data: T[],
  processor: (item: T) => any,
  chunkSize = 100
): Promise<any[]> {
  return new Promise((resolve) => {
    const results: any[] = []
    let index = 0

    function processChunk() {
      const chunk = data.slice(index, index + chunkSize)
      chunk.forEach(item => results.push(processor(item)))
      index += chunkSize

      if (index < data.length) {
        requestAnimationFrame(processChunk)
      } else {
        resolve(results)
      }
    }

    processChunk()
  })
}
```

```typescript
import * as XLSX from 'xlsx'

self.onmessage = (e) => {
  const { file } = e.data

  const reader = new FileReader()
  reader.onload = (event) => {
    const data = event.target.result
    const workbook = XLSX.read(data, { type: 'array' })
    const sheet = workbook.Sheets[workbook.SheetNames[0]]

}
```
